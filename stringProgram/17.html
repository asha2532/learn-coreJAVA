    public static boolean isPalindrome(String str)
    {
    int left=0;
    int rigth=str.length()-1;
    
    while(left<rigth)
    {
    if(str.charAt(left)!=str.charAt(rigth))
    {
    return false;
    }
    
    left++;
    rigth--;
    }
    return true;
    }
    }
