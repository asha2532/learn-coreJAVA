class Keyword
    {
      int x;
    }
    class Usage  extends Keyword
    {
               int x;
      void display()
       {
                 x=120;
              super.x=130;               
          System.out.println("subclass variable :" +x+ "\n superclass variable: " + super.x);
       } 
    }
    class main
    {
     public static void main(String a[])
       {
           Usage obj=new Usage();
                 obj.display();
        }
    }
    

    OUTPUT:
    subclass variable :120
    superclass variable: 130
